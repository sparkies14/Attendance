# Milestone B — Audit Log Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an immutable audit trail for security-relevant actions, an admin UI to view/export/purge it, and close out the legacy cleanup deferred from Milestone A (drop `managers`/`members`, remove `/webhook/check-role`, re-add a smarter credential check).

**Architecture:** Single new `audit_log` table. One pure-logic helper (`lib/audit.js`) handles all writes, fire-and-forget. Each existing route adds one `await audit.log(...)` line per success/failure path. A new `routes/audit.js` exposes `GET /audit` (owner+admin) and `DELETE /audit` (owner only). `admin.html` gains a tab strip and an Audit Log view with filters, CSV export, and a cleanup button gated on a successful export.

**Tech Stack:** Node.js 18+, Express 4, Supabase, bcryptjs, jsonwebtoken, Jest 29.

```js
const supabase = require('./supabase');

const ACTIONS = Object.freeze({
  LOGIN: 'login',
  LOGIN_FAILED: 'login_failed',
  LOGIN_GOOGLE: 'login_google',
  LOGIN_GOOGLE_FAILED: 'login_google_failed',
  REGISTER: 'register',
  PASSWORD_CHANGED: 'password_changed',
  GOOGLE_LINKED: 'google_linked',
  USER_INVITED: 'user_invited',
  USER_PROMOTED: 'user_promoted',
  USER_DEMOTED: 'user_demoted',
  USER_ACTIVATED: 'user_activated',
  USER_DEACTIVATED: 'user_deactivated',
  ATTENDANCE_APPROVED: 'attendance_approved',
  ATTENDANCE_REJECTED: 'attendance_rejected',
  LEAVE_APPROVED: 'leave_approved',
  LEAVE_REJECTED: 'leave_rejected',
  AUDIT_CLEANUP: 'audit_cleanup',
});

function extractIp(req) {
  if (!req || !req.headers) return null;
  const fwd = req.headers['x-forwarded-for'];
  if (fwd) {
    const first = String(fwd).split(',')[0].trim();
    if (first) return first;
  }
  return req.socket?.remoteAddress || null;
}

function extractActor(req, opts) {
  if (opts && opts.actor) return {
    user_id: opts.actor.user_id ?? null,
    email:   opts.actor.email   ?? null,
    role:    opts.actor.role    ?? null,
  };
  if (req && req.user) return {
    user_id: req.user.user_id ?? null,
    email:   req.user.email   ?? null,
    role:    req.user.role    ?? null,
  };
  return { user_id: null, email: null, role: null };
}

async function log(req, action, opts = {}) {
  try {
    const actor = extractActor(req, opts);
    await supabase.from('audit_log').insert({
      actor_user_id:  actor.user_id,
      actor_email:    actor.email,
      actor_role:     actor.role,
      action,
      target_user_id: opts.target_user_id ?? null,
      target_table:   opts.target_table   ?? null,
      target_id:      opts.target_id != null ? String(opts.target_id) : null,
      details:        opts.details        ?? null,
      ip_address:     extractIp(req),
      user_agent:     req?.headers?.['user-agent'] ?? null,
    });
  } catch (err) {
    console.error('Audit log failed:', err.message);
  }
}

module.exports = { log, ACTIONS, extractIp, extractActor };
```

router.post('/change-password', requireAuth, async (req, res) => {
  const { current_password, new_password } = req.body || {};
  if (!validatePassword(new_password)) {
    return res.status(400).json({ error: 'New password must be 8–128 characters.' });
  }

  const { data: user, error } = await supabase

// ─── GET /audit (owner + admin) ──────────────────────────────────────

router.get('/', requireRole('owner', 'admin'), async (req, res) => {
  let page = parseInt(req.query.page);
  if (!Number.isInteger(page) || page < 1) page = 1;
  if (page > MAX_PAGE) page = MAX_PAGE;

  let pageSize = parseInt(req.query.page_size);
  if (!Number.isInteger(pageSize) || pageSize < 1) pageSize = DEFAULT_PAGE_SIZE;
  if (pageSize > MAX_PAGE_SIZE) pageSize = MAX_PAGE_SIZE;

  const { actor, action, from, to } = req.query;

  let q = supabase.from('audit_log')
    .select('*', { count: 'exact' })
    .order('occurred_at', { ascending: false });

  if (actor)  q = q.eq('actor_email', actor);
  if (action) q = q.eq('action', action);
  if (from)   q = q.gte('occurred_at', from);
  if (to)     q = q.lte('occurred_at', to);

  const fromRow = (page - 1) * pageSize;
  const toRow   = fromRow + pageSize - 1;
  q = q.range(fromRow, toRow);

  const { data, count, error } = await q;
  if (error) return res.status(500).json({ error: error.message });

  return res.json({
    page,
    page_size: pageSize,
    total: count ?? 0,
    items: data || [],
  });
});

// ─── DELETE /audit?before=YYYY-MM-DD (owner only) ────────────────────

router.delete('/', requireRole('owner'), async (req, res) => {
  const beforeStr = req.query.before;
  if (!beforeStr) return res.status(400).json({ error: 'before query param required.' });

  const before = new Date(beforeStr);
  if (isNaN(before.getTime())) return res.status(400).json({ error: 'Invalid before date.' });

  const cutoff = new Date(Date.now() - 24 * 3600 * 1000);
  if (before > cutoff) {
    return res.status(400).json({ error: 'before must be at least 24 hours in the past.' });
  }

  // Count first (for the audit_cleanup row's details)
  const { count, error: e1 } = await supabase.from('audit_log')
    .select('id', { count: 'exact', head: true })
    .lt('occurred_at', before.toISOString());
  if (e1) return res.status(500).json({ error: e1.message });

  const { error: e2 } = await supabase.from('audit_log')
    .delete()
    .lt('occurred_at', before.toISOString());
  if (e2) return res.status(500).json({ error: e2.message });

  await audit.log(req, audit.ACTIONS.AUDIT_CLEANUP, {
    details: { before_date: before.toISOString(), rows_deleted: count ?? 0 },
  });

  return res.json({ success: true, rows_deleted: count ?? 0, before: before.toISOString() });
});

  if (action.startsWith('login'))      { bg = 'var(--accent-light)'; color = 'var(--accent)'; }
  if (action.endsWith('_failed'))      { bg = '#fcecec'; color = 'var(--red)'; }
  if (action.startsWith('user_'))      { bg = '#fff4e0'; color = 'var(--amber)'; }
  if (action.includes('approved') || action.includes('rejected')) { bg = '#eef3fa'; color = '#3a5d8c'; }
  if (action === 'audit_cleanup')      { bg = '#f1f1ef'; color = 'var(--text2)'; }
  return `<span class="badge" style="background:${bg};color:${color};">${action}</span>`;
}

function renderAudit(items) {
  const tbody = document.getElementById('audit-tbody');
  if (!items.length) {
    tbody.innerHTML = `<tr><td colspan="5" style="text-align:center;padding:32px;color:var(--text2);">No records.</td></tr>`;
    return;
  }
  tbody.innerHTML = items.map(it => {
    const failed = String(it.action).endsWith('_failed');
    const borderLeft = failed ? 'border-left:3px solid var(--red);' : '';
    const detailsShort = it.details ? escapeHtml(JSON.stringify(it.details)).slice(0, 80) : '—';
    const detailsFull = it.details ? escapeHtml(JSON.stringify(it.details, null, 2)) : '';
    return `
      <tr style="${borderLeft}">
        <td title="${escapeHtml(it.occurred_at)}">${relativeTime(it.occurred_at)}</td>
        <td style="color:var(--text2);">${escapeHtml(it.actor_email || '(anon)')}</td>
        <td>${actionBadge(it.action)}</td>
        <td style="color:var(--text2);font-size:12px;">${escapeHtml(it.target_table || '')}${it.target_id ? '#' + escapeHtml(it.target_id) : ''}</td>
        <td style="color:var(--text2);font-size:12px;font-family:monospace;cursor:pointer;" onclick="showDetailsModal(this)" data-full="${detailsFull}">${detailsShort}${it.details && JSON.stringify(it.details).length > 80 ? '…' : ''}</td>
      </tr>`;
  }).join('');
}

function csvEscape(v) {
  if (v == null) return '';
  const s = typeof v === 'object' ? JSON.stringify(v) : String(v);
  if (/[",\n]/.test(s)) return '"' + s.replace(/"/g, '""') + '"';
  return s;
}

async function exportAuditCsv() {
  const btn = document.getElementById('btn-export');
  btn.disabled = true;
  const origText = btn.textContent;
  btn.textContent = 'Exporting…';

  try {
    // Pull all pages matching the current filters
    const pageSize = 200;
    let page = 1, rows = [], total = Infinity;
    const baseParams = new URLSearchParams({
      page_size: pageSize,
      ...Object.fromEntries(Object.entries(auditState.filters).filter(([_,v]) => v != null && v !== '')),
    });
    while (rows.length < total) {
      const params = new URLSearchParams(baseParams);
      params.set('page', page);
      const res = await apiFetch(`${API_BASE}/audit?${params}`);
      const data = await res.json();
      if (!res.ok) { showStatus('error', data.error || 'Export failed.'); return; }
      total = data.total;
      rows = rows.concat(data.items);
      if (!data.items.length) break;
      page++;
      if (page > 1000) break;  // safety
    }

    const headers = ['occurred_at','actor_email','actor_role','action','target_table','target_id','details','ip_address','user_agent'];
    const csv = [
      headers.join(','),
      ...rows.map(r => headers.map(h => csvEscape(r[h])).join(',')),
    ].join('\n');

    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `audit_log_${new Date().toISOString().slice(0,10)}.csv`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);

    sessionStorage.setItem('anosupo_audit_exported_at', new Date().toISOString());
    refreshCleanButton();
    refreshExportStatus();
    showStatus('success', `Exported ${rows.length} records.`);
  } catch (e) {
    showStatus('error', 'Export failed.');
  } finally {
    btn.disabled = false;
    btn.textContent = origText;
  }
}

function refreshCleanButton() {
  const btn = document.getElementById('btn-clean');
  if (!btn) return;
  const exported = sessionStorage.getItem('anosupo_audit_exported_at') ||
                   sessionStorage.getItem('anosupo_audit_backup_skipped');

}

function refreshExportStatus() {
  const el = document.getElementById('export-status');
  if (!el) return;
  const exportedAt = sessionStorage.getItem('anosupo_audit_exported_at');
  const skipped    = sessionStorage.getItem('anosupo_audit_backup_skipped');
  if (exportedAt) {
    el.textContent = `Last export: ${relativeTime(exportedAt)}`;
    el.style.color = 'var(--accent)';
  } else if (skipped) {
    el.textContent = 'Backup confirmed outside this system.';
    el.style.color = 'var(--amber)';
  } else {
    el.textContent = 'No export this session — please export first.';
    el.style.color = 'var(--text3)';
  }
}

function skipExportGate(ev) {
  ev.preventDefault();
  if (confirm('I confirm I have a recent backup of the audit log outside this system. Proceed?')) {
    sessionStorage.setItem('anosupo_audit_backup_skipped', new Date().toISOString());
    refreshCleanButton();
    refreshExportStatus();
  }
}

let cleanPreviewCount = 0;

function openCleanModal() {
  if (currentUser.role !== 'owner') return;
  document.getElementById('clean-modal').classList.add('show');
  // Default cutoff: 1 year ago
  const d = new Date();
  d.setFullYear(d.getFullYear() - 1);
  document.getElementById('clean-before').value = d.toISOString().slice(0,10);
  document.getElementById('btn-clean-confirm').disabled = true;
  document.getElementById('clean-preview').textContent = '—';
  updateCleanPreview();
  document.getElementById('clean-before').addEventListener('input', updateCleanPreview);
}

function closeCleanModal() {
  document.getElementById('clean-modal').classList.remove('show');
}

async function updateCleanPreview() {
  const before = document.getElementById('clean-before').value;
  if (!before) {
    document.getElementById('clean-preview').textContent = 'Select a date.';
    document.getElementById('btn-clean-confirm').disabled = true;
    return;
  }
  // Use the audit list endpoint with to=before to read total
  try {
    const res = await apiFetch(`${API_BASE}/audit?page=1&page_size=1&to=${encodeURIComponent(before)}`);
    const data = await res.json();
    if (!res.ok) {
      document.getElementById('clean-preview').textContent = data.error || 'Could not preview.';
      document.getElementById('btn-clean-confirm').disabled = true;
      return;
    }
    cleanPreviewCount = data.total;
    document.getElementById('clean-preview').textContent =
      `Will delete ${cleanPreviewCount} record${cleanPreviewCount === 1 ? '' : 's'} older than ${before}.`;
    document.getElementById('btn-clean-confirm').disabled = cleanPreviewCount === 0;
  } catch (e) {
    document.getElementById('clean-preview').textContent = 'Connection failed.';
  }
}

async function confirmClean() {
  const before = document.getElementById('clean-before').value;
  if (!before) return;
  if (!confirm(`Permanently delete ${cleanPreviewCount} records older than ${before}? This cannot be undone.`)) return;
  try {
    const res = await apiFetch(`${API_BASE}/audit?before=${encodeURIComponent(before)}`, { method: 'DELETE' });
    const data = await res.json();
    if (!res.ok) { showStatus('error', data.error || 'Delete failed.'); return; }
    showStatus('success', `Deleted ${data.rows_deleted} record${data.rows_deleted === 1 ? '' : 's'}.`);
    closeCleanModal();
    // Reload list
    auditState.page = 1;
    loadAudit();
    // Reset export gate — user must export again before next cleanup
    sessionStorage.removeItem('anosupo_audit_exported_at');
    sessionStorage.removeItem('anosupo_audit_backup_skipped');
    refreshCleanButton();
    refreshExportStatus();
  } catch (e) {
    showStatus('error', 'Connection failed.');
  }
}
